var sampleApp = angular.module("app",[]);

	sampleApp.controller('sampleControl', function($scope)	{
		$scope.data = [
			{id : "one", content: "one content"},
			{id : "two", content: "two content"},
			{id : "three", content: "three content"}
		]
	});

	sampleApp.directive('list', function()	{
		return {
			scope : {
				listSet : "="
			},
			restrict : "A",
			template : "<ul><li ng-repeat='items in listSet' id='{items.id}' >{items.content}</li></ul>"
		}
	});
